Viewport Height Handling
Mobile browsers shrink usable viewport when address bars hide/show. JS must set:
function updateAppVh() {
  document.documentElement.style.setProperty('--app-vh', `${window.innerHeight}px`);
}
window.addEventListener('resize', updateAppVh);
window.addEventListener('orientationchange', updateAppVh);
updateAppVh();
Use height: var(--app-vh) for full-height containers instead of 100vh.

Run Flow (Async)
High-level recommended pattern:
async function executeRun(input) {
  const controller = new AbortController();
  setRunState({ status:'pending', controller });
  try {
    const safeInput = sanitizeInput(input);
    const result = await fetchRunResult(safeInput, { signal:controller.signal });
    setRunState({ status:'success', data:result });
  } catch (err) {
    if (controller.signal.aborted) {
      setRunState({ status:'aborted' });
    } else {
      logError('executeRun', err);
      setRunState({ status:'error', error:normalizeError(err) });
    }
  } finally {
    scheduleResolverDelay(); // let React settle before backend continues
  }
}

Resolver Delay
function scheduleResolverDelay() {
  // micro-delay to avoid race conditions with backend bridging
  setTimeout(() => {
    // continue or dispatch completion event
  }, 40);
}
